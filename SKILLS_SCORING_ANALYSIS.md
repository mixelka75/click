```python
        return MAX_SCORE  # No requirements

    resume_set = {s.lower().strip() for s in resume_skills}

        (len(matched) / len(required_set)) * 100, 1
    ) if required_set else 0.0

    if matched:
        # –ï—Å—Ç—å —Å–æ–≤–ø–∞–¥–µ–Ω–∏—è - –ø–æ–ª–Ω—ã–π —Ä–∞—Å—á–µ—Ç
        match_ratio = len(matched) / len(required_set)
        return round(match_ratio * MAX_SCORE, 1)

    # –°–ª—É—á–∞–π 3: –ï—Å—Ç—å –Ω–∞–≤—ã–∫–∏, –Ω–æ –Ω–µ —Ç–µ

    # –ß–µ–º –±–æ–ª—å—à–µ –Ω–∞–≤—ã–∫–æ–≤, —Ç–µ–º –≤—ã—à–µ –æ–±—É—á–∞–µ–º–æ—Å—Ç—å
```

```python
def _score_skills_match(
    self,
    resume_skills: List[str],
    required_skills: List[str],
    details: MatchDetails
) -> float:
    MAX_SCORE = 25.0

    if not required_skills:
        return MAX_SCORE

    # Case 1: No skills at all
    if not resume_skills:
        details.skills_missing = required_skills
        details.has_any_skills = False
        return 0.0

    # Normalize
    resume_set = {s.lower().strip() for s in resume_skills}
    required_set = {s.lower().strip() for s in required_skills}

    # Calculate matches
    matched = resume_set.intersection(required_set)
    missing = required_set - matched

    details.skills_matched = sorted(list(matched))
    details.skills_missing = sorted(list(missing))
    details.skills_match_percent = round(
        (len(matched) / len(required_set)) * 100, 1
    ) if required_set else 0.0

    # Case 2: Has matching skills
    if matched:
        match_ratio = len(matched) / len(required_set)
        return round(match_ratio * MAX_SCORE, 1)

    # Case 3: Has other skills (trainability indicator)
    details.has_other_skills = True
    details.other_skills_count = len(resume_skills)

    # Base bonus for having ANY skills (10%)
    base_bonus = MAX_SCORE * 0.10  # 2.5 points

    # Additional bonus based on skill count (up to +10%)
    # More skills = higher trainability
    skill_count_bonus = min(len(resume_skills) / 10.0, 1.0) * (MAX_SCORE * 0.10)

    total_bonus = base_bonus + skill_count_bonus
    return round(total_bonus, 1)
```
