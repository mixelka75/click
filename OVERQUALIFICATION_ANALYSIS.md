```python
    salary_score: float,  # ‚Üê –ù–æ–≤—ã–π –ø–∞—Ä–∞–º–µ—Ç—Ä
    details: MatchDetails
) -> float:
    if candidate_years is None or candidate_years == 0:
        return 0.0

    req_years = self._parse_experience_requirement(required_experience)

    # Check for overqualification (only for junior positions)
    if req_years < 5 and candidate_years > req_years * 3:
        details.overqualified = True

        # If salary doesn't match, strong penalty
        if salary_score < 5.0:
            return 3.0
        # If salary is ok, candidate might be willing
        else:
            return 8.0

    # Normal scoring
    if candidate_years >= req_years:
        details.experience_sufficient = True
        return 10.0
    elif candidate_years >= req_years * 0.75:
        return 7.5
    elif candidate_years >= req_years * 0.5:
        return 5.0
    elif candidate_years >= req_years * 0.25:
        return 2.5
    else:
        return 1.0
```
